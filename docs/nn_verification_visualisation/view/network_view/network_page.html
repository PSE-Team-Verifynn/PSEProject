212            self.__update_display_bounds()

246    def __update_sample_results(self):

248        if index < 0 or index >= len(self.configuration.saved_bounds):
249            self.sample_metrics.set_result(None)
250            self.sample_metrics.setVisible(False)
251            self.full_results_button.setEnabled(False)
252            self.full_results_button.setVisible(False)
253            return

256        # Force a full widget refresh to avoid stale/corrupted summary rendering
257        # when samples are rerun while dialogs are opened/closed.
258        self.sample_metrics.setVisible(False)
259        self.sample_metrics.set_result(None)
260        if result is not None:
261            self.sample_metrics.set_result(result)
262            self.sample_metrics.updateGeometry()
263            self.sample_metrics.update()
264            self.sample_metrics.setVisible(True)
265        self.full_results_button.setEnabled(result is not None)
266        self.full_results_button.setVisible(result is not None)

269        index = self.configuration.selected_bounds_index
270        if index < 0 or index >= len(self.configuration.saved_bounds):
271            return
272        result = self.configuration.saved_bounds[index].get_sample()
273        if result is None:
274            return
275        parent = self.controller.current_network_view
276        parent.open_dialog(SampleResultsDialog(parent.close_dialog, result))

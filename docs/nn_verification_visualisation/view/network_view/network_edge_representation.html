 27    def __init_block_geometry(self):
 28        if not self.source_nodes or not self.target_nodes:
 29            self.polygon = QPolygonF()
 30            self._bounding_rect = QRectF()
 31            return
 32
 33        s_top = self.source_nodes[0].scenePos()
 34        s_bot = self.source_nodes[-1].scenePos()
 35        t_top = self.target_nodes[0].scenePos()
 36        t_bot = self.target_nodes[-1].scenePos()
 37
 38        self.polygon = QPolygonF([s_top, t_top, t_bot, s_bot])
 39        self._bounding_rect = self.polygon.boundingRect()
 40
 41    def __init_line_geometry(self):
 42        self.lines_data = []
 43
 44        xs = self.source_nodes[0].scenePos().x()
 45        xt = self.target_nodes[0].scenePos().x()
 46        min_y = min(self.source_nodes[0].scenePos().y(), self.target_nodes[0].scenePos().y())
 47        max_y = max(self.source_nodes[-1].scenePos().y(), self.target_nodes[-1].scenePos().y())
 48
 49        num_source = len(self.source_nodes)
 50        num_target = len(self.target_nodes)
 51
 52        # Determine the indexing strategy based on shape
 53        transpose_needed = False
 54        if self.weights is not None:
 55            shape = self.weights.shape
 56            # ONNX format is typically [input, output] = [source, target]
 57            if shape == (num_source, num_target):
 58                transpose_needed = False  # Standard format: weights[source, target]
 59            elif shape == (num_target, num_source):
 60                transpose_needed = True  # Transposed format: weights[target, source]
 61
 62        # Build the lines and find weight range
 63        weight_values = []
 64        for i, s in enumerate(self.source_nodes):
 65            for j, t in enumerate(self.target_nodes):
 66                line = QLineF(s.scenePos(), t.scenePos())
 67
 68                w = 1.0
 69                if self.weights is not None:
 70                    try:
 71                        val = self.weights[i, j] if not transpose_needed else self.weights[j, i]
 72                        w = float(val)
 73                        weight_values.append(abs(w))
 74                    except (IndexError, ValueError) as e:
 75                        w = 0.0
 76
 77                self.lines_data.append((line, w))
 78
 79        if weight_values:
 80            self.min_weight = min(weight_values)
 81            self.max_weight = max(weight_values)
 82        else:
 83            self.min_weight = 0.0
 84            self.max_weight = 1.0
 85
 86        self._bounding_rect = QRectF(xs, min_y, xt - xs, max_y - min_y)
 87
 88    def boundingRect(self):
 89        return self._bounding_rect

 91    def paint(self, painter: QPainter, option: QStyleOptionGraphicsItem, widget: QWidget | None = None) -> None:
 92        if self.use_block_mode:
 93            gradient = QLinearGradient(self._bounding_rect.topLeft(), self._bounding_rect.topRight())
 94            gradient.setColorAt(0, QColor(100, 100, 100, 40))
 95            gradient.setColorAt(1, QColor(100, 100, 100, 80))
 96
 97            painter.setBrush(QBrush(gradient))
 98            painter.setPen(Qt.NoPen)
 99            painter.drawPolygon(self.polygon)
100        else:
101            base_color = QColor(150, 150, 150, 150)
102            pen = QPen(base_color)
103            pen.setCosmetic(True)
104
105            # Check if we HAVE weights AND they're valid
106            has_valid_weights = (self.weights is not None and
107                                 self.min_weight is not None and
108                                 len(self.lines_data) > 0)
109
110            if self.use_weighted_mode and has_valid_weights:
111                # WEIGHTED MODE - vary thickness and opacity based on weights
112                min_thickness = 2.3
113                max_thickness = 6.0
114
115                for line, weight in self.lines_data:
116                    abs_weight = abs(weight)
117
118                    # Normalize weight to 0-1 range based on pre-calculated range
119                    if self.max_weight > self.min_weight:
120                        normalized = (abs_weight - self.min_weight) / (self.max_weight - self.min_weight)
121                    else:
122                        normalized = 1.0 if abs_weight > 0 else 0.0
123
124                    # Map to thickness
125                    thickness = min_thickness + normalized * (max_thickness - min_thickness)
126                    pen.setWidthF(thickness)
127
128                    # Alpha based on normalized weight (stronger weights = more opaque)
129                    alpha = int(80 + normalized * 175)  # Range: 80-255
130                    pen.setColor(QColor(100, 100, 100, alpha))
131
132                    painter.setPen(pen)
133                    painter.drawLine(line)
134            else:
135                # NORMAL MODE - uniform lines
136                pen.setWidth(1)
137                pen.setColor(QColor(100, 100, 100, 150))  # Consistent color
138                painter.setPen(pen)
139                lines_only = [x[0] for x in self.lines_data]
140                painter.drawLines(lines_only)

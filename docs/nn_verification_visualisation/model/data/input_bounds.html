 30    def load_bounds(self, bounds: Dict[int, tuple[float, float]]):
 31        '''
 32        Inserts multiple bounds at once by reading them from a dictionary.
 33        Is used by the controller to proces inputs from the InputBoundsLoader.
 34        :param bounds: Bounds to import.
 35        '''
 36        for (key, value) in bounds.items():
 37            if key not in range(self.count):
 38                continue
 39            self.__value[key] = value
 40        # Updates the whole table since every value could have changed. UI subscribes to changes and will update accordingly.
 41        top_left = self.index(0, 0)
 42        bottom_right = self.index(self.rowCount() - 1, self.columnCount() - 1)
 43        self.dataChanged.emit(top_left, bottom_right, self.__ACCEPTED_ROLES)

103    def setData(self, index, value, /, role=Qt.ItemDataRole.EditRole) -> bool | None:
104        '''
105        QAbstractTableModel's data method. Writes the value of a single table cell and calls update events.
106        :param index: index of the cell (tuple)
107        :param value: value to write to the cell
108        :param role: reason for the value write.
109        '''
110        if role != Qt.ItemDataRole.EditRole or not index.isValid():
111            return False
112        # check if the value is actually a float
113        try:
114            value = float(value)
115        except ValueError:
116            return False
117
118        # ensures that the first value never larger than the second value to keep all intervals valid
119        interval = self.__value[index.row()]
120        if index.column() == 0:
121            self.__value[index.row()] = (float(min(value, interval[1])), interval[1])
122        else:
123            self.__value[index.row()] = (interval[0], float(max(value, interval[0])))
124
125        # call the update event of the cell (listened to by the UI)
126        self.dataChanged.emit(index, index, self.__ACCEPTED_ROLES)
127        return True
